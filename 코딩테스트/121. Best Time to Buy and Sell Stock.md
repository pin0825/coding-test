class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        max_p=0
        min_p=float('inf')
        for i in range(len(prices)):
            if min_p>prices[i]:
                min_p=prices[i]
            if (prices[i]-min_p)>=max_p:
                max_p=prices[i]-min_p
        return max_p
